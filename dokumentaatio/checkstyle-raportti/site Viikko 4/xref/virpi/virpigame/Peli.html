25      public void paintComponent(Graphics g) {
26          kartta.virpi().piirraLiikkuva(g);
27          this.sapuska.piirraLiikkuva(g);
28          this.jali.piirraLiikkuva(g);
29          sleepRefresh();
30      }
31  
32      void sleepRefresh() {
33          try {
34              Thread.sleep(500);
35          } catch (Exception e) {
36          }
37          repaint();
38      }
39  
40      /**

42       *
43       * @param kissanNimi annetaan luotavan pelihahmon nimi parametrina
44       */
45      public void aloita(String kissanNimi) {
46          this.lisaaRuokia();
