1   package virpi.virpigame.ui;
2   
3   import java.util.Scanner;
4   import virpi.virpigame.Peli;
5   
6   public class TextUI {
7   
8       private Scanner lukija;
9   
10      public TextUI() {
11          lukija = new Scanner(System.in);
12      }
13  
14      public void start() {
15          Peli peli = new Peli();
16  
17          System.out.println("Tervetuloa kissapeliin Virpi and friends!");
18          System.out.println("Kartalla K-kirjaimet ovat koiria, varo niitä!");
19          System.out.println("Muut kirjaimet ovat ruokia, hahmosi on X");
20          System.out.print("Anna kissalle nimi (tyhjällä Virpi): ");
21          String nimi = lukija.nextLine();
22          if (nimi.equals("")) {
23              peli.aloita("Virpi");
24          } else {
25              peli.aloita(nimi);
26          }
27  
28          //ensimmäiset liikkumiset - pelihahmon liikuttelua ylös ja alas, oikealle ja vasemmalle
29          while (true) {
30              System.out.println("Sinulla on " + peli.getPisteet() + " pistettä");
31              peli.getKartta().tulostaKartta();
32              System.out.println("Liiku ylös, alas, oikea tai vasen (y/a/o/v)");
33              System.out.print("Tyhjällä tai virheellisillä komennoilla pysyt paikallasi, lopeta lopettaa pelin:");
34              String komento = lukija.nextLine();
35              if (komento.equals("lopeta")) {
36                  break;
37              } else if (komento.equals("y") || komento.equals("ylös")) {
38                  peli.getKartta().liikutaHahmoaYlos();
39              } else if (komento.equals("a") || komento.equals("alas")) {
40                  peli.getKartta().liikutaHahmoaAlas();
41              } else if (komento.equals("o") || komento.equals("oikea")) {
42                  peli.getKartta().liikutaHahmoaOikealle();
43              } else if (komento.equals("v") || komento.equals("vasen")) {
44                  peli.getKartta().liikutaHahmoaVasemmalle();
45              }
46              System.out.println("");
47  
48              // kartta päivittyy joka komennon päätteeksi
49              peli.getKartta().paivitaKartta();
50          }
51      }
52  
53  }
