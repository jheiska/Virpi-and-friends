3   import java.awt.Color;
4   import java.awt.Graphics;
5   import java.awt.event.KeyEvent;
6   import java.awt.event.KeyListener;
7   import javax.swing.JPanel;
8   import virpi.virpigame.objects.*;
9   import virpi.virpigame.ui.Alkuruutu;
10  
11  public class Peli extends JPanel implements KeyListener {
12  
13      private Kartta kartta;
14      private Kissa pelihahmo;
15  
16      private enum Pelitila {
17          ALKURUUTU, PELI
18      };
19      private Pelitila tila = Pelitila.ALKURUUTU;
20      private Alkuruutu aloitus;
21  
22      /**
23       * Konstruktorissa luodaan uusi Kartta -olio ja pelihahmo ja aloitetaan
24       * peli.
25       */
26      public Peli() {
27          aloitus = new Alkuruutu();
28          kartta = new Kartta();
29          pelihahmo = new Kissa("Virpi", 0, kartta.haeAloitusRuutu());
30          kartta.lisaaPelihahmo(pelihahmo);
31          this.aloita();
32      }
33  
34      /**
35       * Pelin aloituskomto kutsuu omia koirien ja ruokien lisäilymetodeita.
36       */
37      public void aloita() {
38          this.lisaaRuokia();
39          this.lisaaKoiria();
40  
41      }
42  
43      /**
44       * Tyhjätään, piirretään ja päivitetään kenttää. Mikäli pelitilana on ALKURUUTU, piirretään alkuruutu.
45       * @param g grafiikkaparametri
46       */
47      public void paintComponent(Graphics g) {
48          g.setColor(Color.green);
49          g.fillRect(0, 0, 1200, 240);
50          if (tila.equals(Pelitila.PELI)) {
51              kartta.piirraOliot(g);
52              kartta.paivitaKartta();
53          } else if (tila.equals(Pelitila.ALKURUUTU)) {
54              aloitus.piirraAlkuruutu(g);
55          }
56          sleepRefresh();
57      }
58  
59      /**
60       * Odotellaan ja piirretään uudestaan.
61       */
62      void sleepRefresh() {
63          try {
64              Thread.sleep(300);
65          } catch (Exception e) {
66          }
67          repaint();
68      }
69  
70      public Kartta getKartta() {
71          return kartta;
72      }
73  
74      public int getPisteet() {
75          return this.kartta.getVirpi().getPisteet();
76      }
77  
78      /**
79       * Luodaan ja lisätään muutamia ruoka -olioita kartalle.
80       */
81      public void lisaaRuokia() {
82          Ruoka ruoka1 = new Ruoka("Latz", 100, 3, 3);
83          Ruoka ruoka2 = new Ruoka("Whiskas", 300, 15, 2);
84          Ruoka ruoka3 = new Ruoka("Royal Canin", 500, 22, 0);
85          Ruoka ruoka4 = new Ruoka("Makkara", 500, 30, 4);
86          kartta.lisaaLiikkuva(ruoka1);
87          kartta.lisaaLiikkuva(ruoka2);
88          kartta.lisaaLiikkuva(ruoka3);
89          kartta.lisaaLiikkuva(ruoka4);
90      }
91  
92      /**
93       * Luodaan ja lisätään muutamia koira -olioita kartalle.
94       */
95      public void lisaaKoiria() {
96          Koira k1 = new Koira(-200, 20, 4);
97          Koira k2 = new Koira(-200, 10, 0);
98          Koira k3 = new Koira(-200, 17, 2);
99          Koira k4 = new Koira(-200, 25, 1);
100         kartta.lisaaLiikkuva(k1);
101         kartta.lisaaLiikkuva(k2);
102         kartta.lisaaLiikkuva(k3);
103         kartta.lisaaLiikkuva(k4);
104     }
105 
106     @Override
107     public void keyPressed(KeyEvent ke) {
108         if (tila.equals(Pelitila.ALKURUUTU) && ke.getKeyCode() == KeyEvent.VK_ENTER) {
109             tila = Pelitila.PELI;
110         }
111         if (tila.equals(Pelitila.PELI)) {
112             if (ke.getKeyCode() == KeyEvent.VK_LEFT || ke.getKeyCode() == KeyEvent.VK_A) {
113                 kartta.liikutaHahmoaVasemmalle();
114             } else if (ke.getKeyCode() == KeyEvent.VK_RIGHT || ke.getKeyCode() == KeyEvent.VK_D) {
115                 kartta.liikutaHahmoaOikealle();
116             } else if (ke.getKeyCode() == KeyEvent.VK_UP || ke.getKeyCode() == KeyEvent.VK_W) {
117                 kartta.liikutaHahmoaYlos();
118             } else if (ke.getKeyCode() == KeyEvent.VK_DOWN || ke.getKeyCode() == KeyEvent.VK_S) {
119                 kartta.liikutaHahmoaAlas();
120             }
121         }
122     }
123

125     public void keyTyped(KeyEvent ke) {
126     }
127 
128     @Override
129     public void keyReleased(KeyEvent ke
130     ) {
131     }
